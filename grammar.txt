<program> ::= <declaration>*EOF;
<declaration> ::= <variable>|<statement>;
<variable> ::= "VAR" IDENTIFIER ("=" <expression>)?";";
<statement> ::= <expression>|<output>;
<output> := "OUTPUT "<expression>;
<expression> ::= <assign>|<literal>|<unary>|<binary>|<brackets>;
<literal> ::= INTEGER|STRING|BOOL|NONE;
<brackets> ::= "("<expression>")";
<unary> ::= ("-"|"!")<expression>;
<binary> ::= <expression><operator><expression>;
<operator> ::= "=="|"!="|"<"|"<="|">"|">="|"+"|"-"|"*"|"/";
