<expression> ::= <literal>|<unary>|<binary>|<brackets>;
<literal> ::= INTEGER|STRING|BOOL|NONE;
<brackets> ::= "("<expression>")";
<unary> ::= ("-"|"!")<expression>;
<binary> ::= <expression><operator><expression>;
<operator> ::= "=="|"!="|"<"|"<="|">"|">="|"+"|"-"|"*"|"/";
<expression> ::= <equality>;
<equality> ::= <comparison>(("!="|"==")<comparison>)*;
<comparison> ::= <term>((">"|">="|"<"|"<=")<term>)*;
<term> ::= <factor>(("-"|"+")<factor>)*;
<factor> ::= <unary>(("/","*")<unary>)*;
<unary> ::= ("!"|"-")<unary>|<primary>;
<primary> ::= INTEGER|STRING|BOOL|NONE|"("expression")";
